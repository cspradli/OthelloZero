from OthelloLogic import board
import translate as trans

class OthelloGame{

    oppColor = 0
    color = 0
    board.__init__()

    /*
     *
     * given input from commandline, getInput parses on which command to execute pass/endgame/move/execute
     *
     */
    def getInput(stdin):
        if len(stdin) == 3 or len == 5:
            if stdin[0] == 'W' : oppColor = -1, color = 1
            if stdin[0] == 'B' : oppColor = 1, color = -1
        if len(stdin) == 1:
            if stdin == 'n'
                board.endGame()
            if stdin == 'B' || stdin == 'W'
                #opponent passes
                findMoves(color)
        if len(stdin) == 5:
            x = translateX(stdin[2])
            y = int(stdin[4]) - 1
            board.executeMove(oppColor, (x,y))

    /*
     *
     * Given a color, finds the moves based on board.getValidMoves() function 
     * Outputs command to referee on what move self is makign
     * Ex output B e 4, W a 1, etc
     */
    def findMoves(color):
        out = ''
        if color == 1:
            out.append('W')
        if color == -1:
            out.append('B')
        validMoves = board.getValidMoves(color)
        if len(validMoves) < 0:
            return out
        if len(validMoves) > 0:
            move = validMoves[0]
            board.executeMove(move, color)
            out.append(translateToString(move.x))
            out.append(' ')
            out.append(move.y - 1)
        return out
    /*
     *
     * Translates the input into array-friendly language
     *
     */
    def translateX(inputX):
        switch(inputX):
            case 0:
                outX = 'a'
                break
            case 1:
                outX = 'b'
                break
            case 2:
                outX = 'c'
                break
            case 3:
                outX = 'd'
                break
            case 4:
                outX = 'e'
                break
            case 5:
                outX = 'f'
                break
            case 6:
                outX = 'g'
                break
            case 7:
                outX = 'h'
                break
}
